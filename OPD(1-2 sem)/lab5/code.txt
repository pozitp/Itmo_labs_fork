ORG 0x115
StartAddr : WORD 0x55F
CurAddr: WORD ?
COUNT: WORD 0x0002

START:
	CLA
	LD StartAddr
	ST CurAddr

LOAD:
	LD(CurAddr)+
	CALLING:
		SWAB
		PUSH
		CALL SYMBOL
		POP
		LOOP COUNT
		JUMP CALLING
	JUMP LOAD

STOP: 
	POP
	POP
	HLT

SYMBOL:
	IN 7
	AND #0x40
	BEQ SYMBOL
	LD &1
	SXTB 
	BEQ STOP
	OUT 6
	RET
