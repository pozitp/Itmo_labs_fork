ORG 0x115

StartAddr : WORD 0x55F
CurAddr: WORD ?

Count: WORD 0x0002

Mask: WORD 0x00FF

START:
	CLA
	LD StartAddr
	ST CurAddr

LOADING:
	LD #0x02
	ST Count
	LD(CurAddr)+
	CALLING:
		SWAB
		PUSH
		CALL SYMBOL
		POP
		LOOP Count
		JUMP CALLING
	JUMP LOADING

SYMBOL:
	IN 7
	AND #0x40
	BEQ SYMBOL
	LD &1
	AND MASK 
	BEQ STOP
	OUT 6
	RET

STOP: 
	POP
	POP
	HLT
